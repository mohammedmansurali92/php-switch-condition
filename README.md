# php-switch-condition
<?php
//switch statement
$febcolor = "default";
switch($febcolor){
	case "red";
	echo "My fevorite color is red";
	break;
	case "green";
	echo "My fevorite color is green";
	break;
	case "blue";
	echo "My fevorite color is blue";
	break;
	default:
	echo "My fevorit color neither red, green nor blue";
}
?>
